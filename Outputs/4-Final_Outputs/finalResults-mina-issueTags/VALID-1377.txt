   @Override<br/>    public void
sessionCreated(NextFilter nextFilter,

(!isConnectionOk(session))

}<br/><br/>ConnectionThrottleFilter

throws Exception {<br/>        if
(!isConnectionOk(session)) {           
logger.warn("Connections coming in too
fast; closing.");

nextFilter.sessionCreated(session);<br/>
   }
